          </ImageContainer>
          <Name>{product.name}</Name>
          <Price>{product.price}</Price>

            value={selectedQuantity}
            onChange={handleQuantityChange}

          <AddToCartButton onClick={() => handleAddToCart(product.id)}>
            Add to Cart
          </AddToCartButton>
        </Card>
      ))}
    </ProductContainer>
  );
